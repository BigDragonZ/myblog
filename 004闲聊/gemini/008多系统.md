import os
from bs4 import BeautifulSoup
from markdownify import markdownify as md

def clean_zhihu_html(html_path, output_md_path):
    """将知乎 HTML 清洗并转换为大模型友好的 Markdown 格式"""
    with open(html_path, 'r', encoding='utf-8') as f:
        html_content = f.read()

    # 1. 使用 BeautifulSoup 解析网页树
    soup = BeautifulSoup(html_content, 'html.parser')

    # 2. 定位核心内容区（知乎文章通常在一个特定的 div 类中，如 Post-RichText 或 RichText）
    # 这一步可以帮你过滤掉网页的导航栏、底部的评论区和推荐广告
    article_body = soup.find('div', class_='RichText') 
    
    if not article_body:
        print(f"警告: 在 {html_path} 中未找到正文，将转换整个页面。")
        article_body = soup # 如果没找到，退一步转换全部

    # 3. 剥离无用的脚本和样式
    for script_or_style in article_body(['script', 'style']):
        script_or_style.extract()

    # 4. 转换为干净的 Markdown
    md_text = md(str(article_body), heading_style="ATX")

    # 5. 保存结果
    with open(output_md_path, 'w', encoding='utf-8') as f:
        f.write(md_text)
    
    print(f"✅ 成功转换: {output_md_path}")

### 3. 编写核心大脑调度器 (The Agent Executor)

这是系统的主干。它负责读取你的（未来填入的）`brain_prompt.txt`，并决定何时调用上述工具。

**`main_agent.py`**

Python

```
import os
from langchain.agents import create_tool_calling_agent, AgentExecutor
from langchain_core.prompts import ChatPromptTemplate
# 这里假设未来你在 Windows 上本地部署了 Qwen 或 Llama 模型
# 演示暂时使用 langchain 的通用大模型接口占位
from langchain_community.llms import FakeListLLM 

# 1. 导入你写好的“空壳”技能
from skills.tool_finance import get_stock_price
from skills.tool_rag_search import query_investment_principles

def load_brain_prompt():
    """读取预留的价值观提示词"""
    prompt_path = "config/brain_prompt.txt"
    if not os.path.exists(prompt_path) or os.path.getsize(prompt_path) == 0:
        return "你是一个价值投资助理。请使用可用工具分析股票。" # 默认的底线 Prompt
    with open(prompt_path, "r", encoding="utf-8") as f:
        return f.read()

def run_agent_pipeline(target_stock: str):
    print(f"=== 启动 ValueAgent，开始分析 {target_stock} ===")
    
    # 2. 组装工具箱
    tools = [get_stock_price, query_investment_principles]
    
    # 3. 构建大脑 Prompt
    system_instruction = load_brain_prompt()
    prompt = ChatPromptTemplate.from_messages([
        ("system", system_instruction),
        ("human", "{input}"),
        ("placeholder", "{agent_scratchpad}"),
    ])
    
    # 4. 初始化本地大模型 (这里用 Fake LLM 占位，防止未配置环境时报错)
    # TODO: 未来在 WSL2 中替换为 Ollama 或 vLLM 驱动的 RTX 4060 本地模型
    llm = FakeListLLM(responses=["我已经查阅了价格和原则，建议持续关注。"]) 
    
    # 5. 绑定 Agent 和执行器
    agent = create_tool_calling_agent(llm, tools, prompt)
    agent_executor = AgentExecutor(agent=agent, tools=tools, verbose=True)
    
    # 6. 给 Agent 下达首个任务
    task = f"请帮我全面评估 {target_stock} 目前是否具备投资价值，并在必要时设置监控点位。"
    response = agent_executor.invoke({"input": task})
    
    print("\n=== Agent 最终结论 ===")
    print(response['output'])

if __name__ == "__main__":
    # 测试一只股票
    run_agent_pipeline("0700.HK")
```
